# Fundamentals - Spacing

We are currently using the default values for spacing. These values are used for all margins, paddings, widths and heights.

```html
<div class="tw-mt-1">This div will have a `margin-top` of 4 pixels</div>
```


<Canvas>
  <Story name="Spacings">
    {()=> ({
      data() {
        return {
          unit: 4,
          spacings: [ 0, 'px', 0.5, 1, 1.5, 2, 2.5, 3, 3.5, 4, 5, 6, 7, 8, 9, 10, 11, 12, 14, 16, 20, 24, 28, 32, 36, 40, 44, 48, 52, 56, 60, 64, 72, 80, 96 ]
        }
            },
            methods: {
                rem(value) {
                    return value === 'px' ? '1px' : `${_.round(value / this.unit, 3)}rem`;
                },
                px(value) {
                    return value === 'px' ? '1px' : `${_.round(value * this.unit, 3)}px`;
                },
            },
            template: `
            <table>
              <thead>
                <tr class='tw-text-left'>
                  <th class="tw-px-2">Name</th>
                  <th class="tw-px-2">Size</th>
                  <th class="tw-px-2">Pixels</th>
                  <th class="hidden sm:table-cell">
                    <span class="sr-only">Preview</span>
                  </th>
                </tr>
              </thead>
              <tbody>
                <tr v-for="spacing in spacings">
                  <td class="tw-px-2">{{spacing}}</td>
                  <td class="tw-px-2">{{rem(spacing)}}</td>
                  <td class="tw-px-2">{{px(spacing)}}</td>
                  <td class="tw-hidden sm:tw-table-cell">
                    <div class="tw-h-4 tw-bg-violet-400" :style="{width: px(spacing)}"></div>
                  </td>
                </tr>
              </tbody>
            </table>
          `,
        })}
    </Story>
</Canvas>
